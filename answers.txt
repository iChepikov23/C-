from selenium import webdriver

def main():
    driver = webdriver.Chrome()
    driver.get("https://app.tetrainsights.com/projects")
    elem = driver.find_element_by_class_name("tetra-btn btn m-0 mb-2 btn-primary tetra-btn-container btn-lg btn-block")
    print(elem)
    elem.click()


if __name__ == "__main__":
    main()
